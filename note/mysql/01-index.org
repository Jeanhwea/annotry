* 数据准备
  添加 author 表
  #+BEGIN_SRC sql
    drop table author;

    create table author (
      id int auto_increment primary key,
      name varchar(128) not null,
      age int,
      gender varchar(32),
      a int,
      b int,
      c int
    );
  #+END_SRC
